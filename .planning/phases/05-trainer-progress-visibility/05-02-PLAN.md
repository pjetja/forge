**Data fetching (same logic as trainer version, scoped to claims.sub):**

1. Get claims:
```typescript
const supabase = await createClient();
const claimsResult = await supabase.auth.getClaims();
const claims = claimsResult.data?.claims;
if (!claims) redirect('/login');
```

2. Fetch exercise name via admin client:
```typescript
const admin = createAdminClient();
const { data: exerciseRow } = await admin
  .from('exercises')
  .select('name, muscle_group')
  .eq('id', exerciseId)
  .single();
if (!exerciseRow) notFound();
```

3. Get all ASE IDs for this base exercise_id:
```typescript
const { data: aseRows } = await supabase
  .from('assigned_schema_exercises')
  .select('id')
  .eq('exercise_id', exerciseId);
const aseIds = (aseRows ?? []).map(r => r.id);
```

4. Get all completed sessions for this trainee:
```typescript
const { data: sessionsData } = await supabase
  .from('workout_sessions')
  .select('id, completed_at')
  .eq('trainee_auth_uid', claims.sub)
  .eq('status', 'completed')
  .order('completed_at', { ascending: true });
```
