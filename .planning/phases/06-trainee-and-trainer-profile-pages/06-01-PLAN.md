    **2. Create `src/lib/gravatar.ts`:**

    Server-only utility using Node.js built-in `crypto` module (NOT Web Crypto API, which does not support MD5).

    ```typescript
    import { createHash } from 'crypto';

    export function gravatarUrl(email: string, size = 80): string {
      const hash = createHash('md5').update(email.trim().toLowerCase()).digest('hex');
      return `https://www.gravatar.com/avatar/${hash}?s=${size}&d=mp`;
    }
    ```

    **3. Create `src/components/GravatarAvatar.tsx`:**

    Shared component, NOT a client component (no 'use client' directive — it's a pure presentational component that works in both server and client contexts).

    ```tsx
    interface GravatarAvatarProps {
      url: string;
      name: string;
      size?: number;
      className?: string;
    }

    export function GravatarAvatar({ url, name, size = 40, className }: GravatarAvatarProps) {
      return (
        // eslint-disable-next-line @next/next/no-img-element
        <img
          src={url}
          alt={`${name}'s avatar`}
          width={size}
          height={size}
          className={`rounded-full object-cover bg-accent/20 ${className ?? ''}`}
        />
      );
    }
    ```

    Use plain `<img>` to avoid needing `remotePatterns` config for gravatar.com. The `d=mp` query param ensures a valid silhouette image always loads.
