  <action>
    Replace the entire contents of globals.css with the following:

    ```css
    @import "tailwindcss";

    @theme inline {
      /* Font — next/font injects --font-lato onto <html> via className={lato.variable} */
      --font-sans: var(--font-lato);

      /* Design system colors */
      --color-bg-page:      #0f172a;
      --color-bg-surface:   #1e293b;
      --color-border:       #334155;
      --color-accent:       #10b981;
      --color-accent-hover: #34d399;
      --color-text-primary: #e2e8f0;
      --color-error:        #ef4444;
      --color-error-light:  #f87171;
    }

    @layer base {
      body {
        background-color: #0f172a;
        color: #e2e8f0;
      }
    }
    ```

    Remove: the old `:root` block, the `@media (prefers-color-scheme: dark)` block, and the standalone `body {}` rule. The `@layer base` block replaces all of these.

    Use `@theme inline` (not plain `@theme`) because `--font-sans` references a CSS custom property (`var(--font-lato)`). Plain `@theme` would create broken chained variable resolution.
  </action>

  </verify>

<task type="auto">
  <name>Task 2: Replace Geist with Lato in root layout + restyle auth layout</name>
  <files>src/app/layout.tsx, src/app/(auth)/layout.tsx</files>
  <action>
    **src/app/layout.tsx** — replace entire file:

    ```tsx
    import type { Metadata } from 'next';
    import { Lato } from 'next/font/google';
    import './globals.css';

    const lato = Lato({
      variable: '--font-lato',
      subsets: ['latin'],
      weight: ['400', '700'],
      display: 'swap',
    });

    export const metadata: Metadata = {
      title: 'Forge',
      description: 'Track your training. Forge your progress.',
    };

    export default function RootLayout({ children }: { children: React.ReactNode }) {
      return (
        <html lang="en" className={lato.variable}>
          <body className="antialiased">
            {children}
          </body>
        </html>
      );
    }
    ```

    IMPORTANT: Use `variable: '--font-lato'` and add the variable class to `<html>`, NOT to `<body>`. This injects the CSS custom property at the html element scope so globals.css `@theme inline` can resolve `var(--font-lato)`. Lato is not a variable font — `weight: ['400', '700']` is required (omitting it causes a Next.js build error).

    **src/app/(auth)/layout.tsx** — replace entire file:

    ```tsx
    export default function AuthLayout({ children }: { children: React.ReactNode }) {

          <div className="w-full max-w-md px-4">{children}</div>
        </div>
      );
    }
    ```

    Change: `bg-gray-50` → `bg-bg-page`. Add `px-4` to inner wrapper for mobile padding. No other changes needed.
