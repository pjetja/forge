    - Muscle group (required — render as a <select> dropdown with all 13 MUSCLE_GROUPS options, not chips — chips are for the filter bar)
    - Description (optional textarea, max 500 chars)
    - Notes (optional textarea, max 500 chars, label "Coaching notes")
    - Video URL (optional text input, label "Video URL (YouTube)")

    Zod schema:
    ```typescript
    import { z } from 'zod';
    import { MUSCLE_GROUPS } from '@/lib/db/schema';

    function extractYouTubeId(url: string): string | null {
      const regex = /(?:youtube(?:-nocookie)?\.com\/(?:[^\\/\n\s]+\/\S+\/|(?:v|e(?:mbed)?)\/|\S*?[?&]v=)|youtu\.be\/|youtube\.com\/shorts\/)([a-zA-Z0-9_-]{11})/;
      const match = url.match(regex);
      return match?.[1] ?? null;
    }

    const exerciseSchema = z.object({
      name: z.string().min(1, 'Exercise name is required').max(100, 'Name too long'),
      muscleGroup: z.enum(MUSCLE_GROUPS, { error: 'Select a muscle group' }),
      description: z.string().max(500, 'Max 500 characters').optional(),
      notes: z.string().max(500, 'Max 500 characters').optional(),
      videoUrl: z
        .string()
        .optional()
        .refine(
          (val) => !val || extractYouTubeId(val) !== null,
          'Must be a valid YouTube URL (or leave blank)'
        ),
    });
    ```

    Note: Zod v4 uses .issues (not .errors) on ZodError — established Phase 1 decision.
