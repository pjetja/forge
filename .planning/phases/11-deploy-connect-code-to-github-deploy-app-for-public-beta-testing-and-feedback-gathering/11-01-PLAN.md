  <name>Task 4: Generate PNG icons from src/app/icon.svg</name>
  <what-needed>
    The PWA manifest requires `public/icon-192.png` (192×192) and `public/icon-512.png` (512×512).
    The source is `src/app/icon.svg`.

    Claude cannot generate binary PNG files. Export them using one of these methods:

    **Option A — Browser (fastest, no install needed):**
    1. Open `src/app/icon.svg` in Chrome/Firefox
    2. Use https://svgtopng.com or https://convertio.co/svg-png/
    3. Export at 192×192 → save as `public/icon-192.png`
    4. Export at 512×512 → save as `public/icon-512.png`

    **Option B — Inkscape (if installed):**
    ```bash
    inkscape src/app/icon.svg --export-type=png --export-filename=public/icon-192.png --export-width=192 --export-height=192
    inkscape src/app/icon.svg --export-type=png --export-filename=public/icon-512.png --export-width=512 --export-height=512
    ```

    **Option C — Node sharp (install once):**
    ```bash
    pnpm add -D sharp
    node -e "
      const sharp = require('sharp');
      sharp('src/app/icon.svg').resize(192).png().toFile('public/icon-192.png');
      sharp('src/app/icon.svg').resize(512).png().toFile('public/icon-512.png');
    "
    ```

    Place the two PNG files at `public/icon-192.png` and `public/icon-512.png`.

  </what-needed>

    ```typescript
    import type { Metadata } from 'next';
    import { Lato } from 'next/font/google';
    import NextTopLoader from 'nextjs-toploader';
    import { NextIntlClientProvider } from 'next-intl';
    import { getLocale } from 'next-intl/server';
    import './globals.css';

    const lato = Lato({
      variable: '--font-lato',
      subsets: ['latin', 'latin-ext'],
      weight: ['400', '700'],
      display: 'swap',
    });

    export const metadata: Metadata = {
      title: 'Forge',
      description: 'Track your training. Forge your progress.',
    };

    export default async function RootLayout({ children }: { children: React.ReactNode }) {
      const locale = await getLocale();
      return (
        <html lang={locale} className={lato.variable}>
          <head>
            <link rel="manifest" href="/manifest.json" />
            <meta name="theme-color" content="#0f172a" />
            <meta name="apple-mobile-web-app-capable" content="yes" />
            <meta name="apple-mobile-web-app-status-bar-style" content="black-translucent" />
            <link rel="apple-touch-icon" href="/icon-192.png" />
          </head>
          <body className="antialiased">
            <NextTopLoader color="#4ade80" shadow={false} showSpinner={false} height={2} />
            <NextIntlClientProvider>{children}</NextIntlClientProvider>
          </body>
        </html>
      );
    }
    ```
