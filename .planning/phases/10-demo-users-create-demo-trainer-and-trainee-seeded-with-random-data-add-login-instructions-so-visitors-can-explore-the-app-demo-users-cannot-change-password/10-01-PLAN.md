// Date helper — today is 2026-03-30 but script should work at any run date
function daysAgo(n: number): string {
  const d = new Date();
  d.setDate(d.getDate() - n);
  return d.toISOString(); // for timestamps
}
function daysAgoDate(n: number): string {
  const d = new Date();
  d.setDate(d.getDate() - n);
  return d.toISOString().split("T")[0]; // 'YYYY-MM-DD' for logged_date
}
